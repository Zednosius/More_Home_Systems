# Runs the durin event on game start to set the flag to its owner
on_game_start = {
    events = {
        mhs_durin.1
    }
}

# Opens a Popup when the homesystem survey is finished in the durin event
# A ship is done surveying the last unsurveyed planet in a system
# Scope = ship
# From = system
on_system_survey = {
    events = {
        mhs_durin.2

    }
}
# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey = {
    events = {
        mhs_durin.5
    }
}

# Triggers event when ship enters a system
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system = {
    events = {
        mhs_durin.3
    }
}

# A planet has been colonized.
# Scope = Planet
on_colonized = {
    events = {
        mhs_durin.9
    }
}

# From = Tile
# This = Planet
on_blocker_cleared = {
    events = {
        mhs_durin.10
    }
}

# Triggers country_event for the attacker upon victory
# This = country, leader attacker
# From = country, planet owner
# FromFrom = planet
on_planet_attackers_win = {
    events = {
        mhs_durin.13
        mhs_durin.16
    }
}

# Triggers country_event for the defender upon victory
# This = country, planet owner
# From = country, attack leader
# FromFrom = planet
on_planet_defenders_win = {
    events = {
        mhs_durin.15
    }
}
