# Runs the durin event on game start to set the flag to its owner
on_game_start = {
    events = {
        mhs_durin.1
    }
}

# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey = {
    events = {
        mhs_durin.4
    }
}

# Triggers event when ship enters a system
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system = {
    events = {
        mhs_durin.3
    }
}

# A planet has been colonized.
# Scope = Planet

	events = {
		mhs_durin.7
	}
}

# This = Planet

    events = {

    }
}
