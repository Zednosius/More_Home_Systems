# Runs the durin event on game start to set the flag to its owner
on_game_start = {
    events = {
        mhs_durin.1
    }
}

# Opens a Popup when the homesystem survey is finished in the durin event
# A ship is done surveying the last unsurveyed planet in a system
# Scope = ship
# From = system
on_system_survey = {
    events = {
        mhs_durin.2

    }
}
# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey = {
    events = {
        mhs_durin.5
    }
}

# Triggers event when ship enters a system
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system = {
    events = {
        mhs_durin.4
    }
}

# A planet has been colonized.
# Scope = Planet
on_colonized = {
    events = {
        mhs_durin.9
    }
}

# From = Tile
# This = Planet
on_blocker_cleared = {
    events = {
        mhs_durin.10
    }
}
